277 // Scan a page to determine how many words have been written to it.
278 // This information is used to set the page write offset during initialization.
279 // Additionally, this function updates the latest record ID as it proceeds.
280 // If an invalid record header is found, the can_gc argument is set to true.
281 static void page_scan (uint32_t const * p_addr,
282  uint16_t * const words_written,
283  bool * const can_gc)
284 {
285  uint32_t const * const p_page_end = p_addr + FDS_PAGE_SIZE;
286  p_addr += FDS_PAGE_TAG_SIZE;
287  *words_written = FDS_PAGE_TAG_SIZE;
288  fds_header_t const * p_header = (fds_header_t *) p_addr;
289 
290  while (header_has_next (p_header, p_page_end))
291  {
292  fds_header_status_t hdr = header_check (p_header, p_page_end);
293 
294  if (hdr == FDS_HEADER_VALID)
295  {
296  // Update the latest (largest) record ID.
297  if (p_header->record_id > m_latest_rec_id)
298  {
299  m_latest_rec_id = p_header->record_id;
300  }
301  }
302  else
303  {
304  if (can_gc != NULL)
305  {
306  *can_gc = true;
307  }
308 
309  if (hdr == FDS_HEADER_CORRUPT)
310  {
311  // It could happen that a record has a corrupt header which would set a
312  // wrong offset for this page. In such cases, update this value to its maximum,
313  // to ensure that no new records will be written to this page and to enable
314  // correct statistics reporting by fds_stat().
315  *words_written = FDS_PAGE_SIZE;
316  // We can't continue to scan this page.
317  return;
318  }
319  }
320 
321  *words_written += (FDS_HEADER_SIZE + p_header->length_words);
322  p_header = header_jump (p_header);
323  }
324 }

328 {
329  // If the first part of the header has been written correctly, update the offset as normal.

363 // Reserve space on a page.
364 // NOTE: this function takes into the account the space required for the record header.
365 static ret_code_t write_space_reserve (uint16_t length_words, uint16_t * p_page)
366 {
367  bool space_reserved = false;
368  uint16_t const total_len_words = length_words + FDS_HEADER_SIZE;
369 
370  if (total_len_words >= FDS_PAGE_SIZE - FDS_PAGE_TAG_SIZE)
371  {
372  return FDS_ERR_RECORD_TOO_LARGE;
373  }
374 
375  CRITICAL_SECTION_ENTER();
376 
377  for (uint16_t page = 0; page < FDS_DATA_PAGES; page++)
378  {
379  if ( (m_pages[page].page_type == FDS_PAGE_DATA) &&
380  (page_has_space (page, total_len_words)))
381  {
382  space_reserved = true;
383  *p_page = page;
384  m_pages[page].words_reserved += total_len_words;

408 // Given a page and a record, find the next valid record on that page.
409 // If p_record is NULL, search from the beginning of the page,
410 // otherwise, resume searching from p_record.
411 // Return true if a record is found, false otherwise.
412 // If no record is found, p_record is unchanged.
413 static bool record_find_next (uint16_t page, uint32_t const ** p_record)
414 {
415  uint32_t const * p_page_end = (m_pages[page].p_addr + FDS_PAGE_SIZE);
416  // If this is the first call on this page, start searching from its beginning.
417  // Otherwise, jump to the next record.
418  fds_header_t const * p_header = (fds_header_t *) (*p_record);
419 
420  if (p_header != NULL)
421  {
422  p_header = header_jump (p_header);
423  }
424  else
425  {
426  p_header = (fds_header_t *) (m_pages[page].p_addr + FDS_PAGE_TAG_SIZE);
427  }
428 
429  // Read records from the page until:
430  // - a valid record is found or
431  // - the last record on a page is found
432 
433  while (header_has_next (p_header, p_page_end))
434  {
435  switch (header_check (p_header, p_page_end))
436  {
437  case FDS_HEADER_VALID:
438  *p_record = (uint32_t *) p_header;
439  return true;
440 
441  case FDS_HEADER_DIRTY:
442  p_header = header_jump (p_header);
443  break;
444 
445  case FDS_HEADER_CORRUPT:
446  // We can't reliably jump over this record.
447  // There is nothing more we can do on this page.
448  return false;
449  }
450  }
451 
452  // No more valid records on this page.
453  return false;
454 }

457 // Find a record given its descriptor and retrive the page in which the record is stored.
458 // NOTE: Do not pass NULL as an argument for p_page.
459 static bool record_find_by_desc (fds_record_desc_t * const p_desc,
460  uint16_t * const p_page)
461 {
462  // If the gc_run_count field in the descriptor matches our counter, then the record has
463  // not been moved. If the address is valid, and the record ID matches, there is no need
464  // to find the record again. Only lookup the page in which the record is stored.
465  if ( (address_is_valid (p_desc->p_record)) &&
466  (p_desc->gc_run_count == m_gc.run_count) &&
467  (p_desc->record_id == ( (fds_header_t *) p_desc->p_record)->record_id))
468  {
469  return (page_from_record (p_page, p_desc->p_record) == FDS_SUCCESS);
470  }
471 
472  // Otherwise, find the record in flash.
473  for (*p_page = 0; *p_page < FDS_DATA_PAGES; (*p_page)++)
474  {
475  // Set p_record to NULL to make record_find_next() search from the beginning of the page.
476  uint32_t const * p_record = NULL;
477 
478  while (record_find_next (*p_page, &p_record))
479  {
480  fds_header_t const * const p_header = (fds_header_t *) p_record;
481 
482  if (p_header->record_id == p_desc->record_id)
483  {
484  p_desc->p_record = p_record;
485  p_desc->gc_run_count = m_gc.run_count;
486  return true;
487  }
488  }
489  }
490 
491  return false;
492 }

495 // Search for a record and return its descriptor.
496 // If p_file_id is NULL, only the record key will be used for matching.
497 // If p_record_key is NULL, only the file ID will be used for matching.
498 // If both are NULL, it will iterate through all records.
499 static ret_code_t record_find (uint16_t const * p_file_id,
500  uint16_t const * p_record_key,
501  fds_record_desc_t * p_desc,
502  fds_find_token_t * p_token)
503 {
504  if (!m_flags.initialized)
505  {
506  return FDS_ERR_NOT_INITIALIZED;
507  }
508 
509  if (p_desc == NULL || p_token == NULL)
510  {
511  return FDS_ERR_NULL_ARG;
512  }
513 
514  // Begin (or resume) searching for a record.
515  for (; p_token->page < FDS_DATA_PAGES; p_token->page++)
516  {
517  if (m_pages[p_token->page].page_type != FDS_PAGE_DATA)
518  {
519  // It might be that the page is FDS_PAGE_UNDEFINED.
520  // Skip this page.
521  continue;
522  }
523 
524  while (record_find_next (p_token->page, &p_token->p_addr))
525  {
526  fds_header_t const * p_header = (fds_header_t *) p_token->p_addr;
527 
528  // A valid record was found, check its header for a match.
529  if ( (p_file_id != NULL) &&
530  (p_header->file_id != *p_file_id))
531  {
532  continue;
533  }
534 
535  if ( (p_record_key != NULL) &&
536  (p_header->record_key != *p_record_key))
537  {
538  continue;
539  }
540 
541  // Record found; update the descriptor.
542  p_desc->record_id = p_header->record_id;
543  p_desc->p_record = p_token->p_addr;
544  p_desc->gc_run_count = m_gc.run_count;
545  return FDS_SUCCESS;
546  }
547 
548  // We have scanned an entire page. Set the address in the token to NULL
549  // so that it will be updated in the next iteration.
550  p_token->p_addr = NULL;
551  }
552 
553  return FDS_ERR_NOT_FOUND;
554 }

558 static void records_stat (uint16_t page,
559  uint16_t * p_valid_records,
560  uint16_t * p_dirty_records,
561  uint16_t * p_freeable_words,

638 // This function is called during initialization to setup the page structure (m_pages) and
639 // provide additional information regarding eventual further initialization steps.
640 static fds_init_opts_t pages_init (void)
641 {
642  uint32_t ret = NO_PAGES;
643  uint16_t page = 0;
644  uint16_t total_pages_available = FDS_VIRTUAL_PAGES;
645  bool swap_set_but_not_found = false;
646 
647  for (uint16_t i = 0; i < FDS_VIRTUAL_PAGES; i++)
648  {
649  uint32_t const * const p_page_addr = (uint32_t *) m_fs.start_addr +
650  (i * FDS_PAGE_SIZE);
651  fds_page_type_t const page_type = page_identify (p_page_addr);
652 
653  switch (page_type)
654  {
655  case FDS_PAGE_UNDEFINED:
656  {

658  {

662  // and try to tag it as data (in flash) later on during initialization.
663  m_pages[page].page_type = FDS_PAGE_ERASED;
664  m_pages[page].p_addr = p_page_addr;
665  m_pages[page].write_offset = FDS_PAGE_TAG_SIZE;
666  // This is a candidate for a potential new swap page, in case the
667  // current swap is going to be promoted to complete a GC instance.
668  m_gc.cur_page = page;
669  page++;
670  }
671  else
672  {
673  // If there is no swap page yet, use this one.
674  m_swap_page.p_addr = p_page_addr;
675  m_swap_page.write_offset = FDS_PAGE_TAG_SIZE;
676  swap_set_but_not_found = true;
677  }
678 
679  ret |= PAGE_ERASED;
680  }
681  else
682  {
683  // The page contains non-FDS data.
684  // Do not initialize or use this page.
685  total_pages_available--;
686  m_pages[page].p_addr = p_page_addr;
687  m_pages[page].page_type = FDS_PAGE_UNDEFINED;
688  page++;
689  }
690  }
691  break;
692 
693  case FDS_PAGE_DATA:
694  {
695  m_pages[page].page_type = FDS_PAGE_DATA;
696  m_pages[page].p_addr = p_page_addr;
697  // Scan the page to compute its write offset and determine whether or not the page
698  // can be garbage collected. Additionally, update the latest kwown record ID.
699  page_scan (p_page_addr, &m_pages[page].write_offset, &m_pages[page].can_gc);
700  ret |= PAGE_DATA;
701  page++;
702  }
703  break;
704 
705  case FDS_PAGE_SWAP:
706  {
707  if (swap_set_but_not_found)
708  {
709  m_pages[page].page_type = FDS_PAGE_ERASED;
710  m_pages[page].p_addr = m_swap_page.p_addr;
711  m_pages[page].write_offset = FDS_PAGE_TAG_SIZE;
712  page++;
713  }
714 
715  m_swap_page.p_addr = p_page_addr;
716  // If the swap is promoted, this offset should be kept, otherwise,
717  // it should be set to FDS_PAGE_TAG_SIZE.
718  page_scan (p_page_addr, &m_swap_page.write_offset, NULL);
719  ret |= (m_swap_page.write_offset == FDS_PAGE_TAG_SIZE) ?
720  PAGE_SWAP_CLEAN : PAGE_SWAP_DIRTY;
721  }
722  break;
723 
724  default:
725  // Shouldn't happen.
726  break;
727  }
728  }
729 
730  if (total_pages_available < 2)
731  {
732  ret &= NO_PAGES;
733  }
734 
735  return (fds_init_opts_t) ret;
736 }

866 static bool crc_verify_success (uint16_t crc, uint16_t len_words,
867  uint32_t const * const p_data)
868 {

875  computed_crc = crc16_compute ( (uint8_t const *) p_data + 8,
876  (FDS_HEADER_SIZE_ID + len_words) * sizeof (uint32_t),

879 }

1087 // Initialize the filesystem.
1088 static ret_code_t init_execute (uint32_t prev_ret, fds_op_t * const p_op)
1089 {
1090  ret_code_t ret = FDS_ERR_INTERNAL;
1091 
1092  if (prev_ret != NRF_SUCCESS)
1093  {
1094  // A previous operation has timed out.
1095  m_flags.initializing = false;
1096  return FDS_ERR_OPERATION_TIMEOUT;
1097  }
1098 
1099  switch (p_op->init.step)
1100  {
1101  case FDS_OP_INIT_TAG_SWAP:
1102  {
1103  // The page write offset was determined previously by pages_init().
1104  p_op->init.step = FDS_OP_INIT_TAG_DATA;
1105  ret = page_tag_write_swap();
1106  }
1107  break;
1108 
1109  case FDS_OP_INIT_TAG_DATA:
1110  {
1111  // Tag remaining erased pages as data.
1112  bool write_reqd = false;
1113 
1114  for (uint16_t i = 0; i < FDS_DATA_PAGES; i++)
1115  {
1116  if (m_pages[i].page_type == FDS_PAGE_ERASED)
1117  {
1118  m_pages[i].page_type = FDS_PAGE_DATA;
1119  write_reqd = true;
1120  ret = page_tag_write_data (m_pages[i].p_addr);
1121  break;
1122  }
1123  }
1124 
1125  if (!write_reqd)
1126  {
1127  m_flags.initialized = true;
1128  m_flags.initializing = false;
1129  return FDS_OP_COMPLETED;
1130  }
1131  }
1132  break;
1133 
1134  case FDS_OP_INIT_ERASE_SWAP:
1135  {
1136  // If the swap is going to be discarded then reset its write_offset.
1137  p_op->init.step = FDS_OP_INIT_TAG_SWAP;
1138  m_swap_page.write_offset = FDS_PAGE_TAG_SIZE;
1139  ret = nrf_fstorage_erase (&m_fs, (uint32_t) m_swap_page.p_addr, FDS_PHY_PAGES_IN_VPAGE,
1140  NULL);
1141  }
1142  break;
1143 
1144  case FDS_OP_INIT_PROMOTE_SWAP:
1145  {
1146  p_op->init.step = FDS_OP_INIT_TAG_SWAP;
1147  // When promoting the swap, keep the write_offset set by pages_init().
1148  ret = page_tag_write_data (m_swap_page.p_addr);
1149  uint16_t const gc = m_gc.cur_page;
1150  uint32_t const * const p_old_swap = m_swap_page.p_addr;
1151  // Execute the swap.
1152  m_swap_page.p_addr = m_pages[gc].p_addr;
1153  m_pages[gc].p_addr = p_old_swap;
1154  // Copy the offset from the swap to the new page.
1155  m_pages[gc].write_offset = m_swap_page.write_offset;
1156  m_swap_page.write_offset = FDS_PAGE_TAG_SIZE;
1157  m_pages[gc].page_type = FDS_PAGE_DATA;
1158  }
1159  break;
1160 
1161  default:
1162  // Should not happen.
1163  break;
1164  }
1165 
1166  if (ret != FDS_SUCCESS)
1167  {
1168  // fstorage queue was full.
1169  m_flags.initializing = false;
1170  return FDS_ERR_BUSY;
1171  }
1172 
1173  return FDS_OP_EXECUTING;
1174 }

1177 // Executes write and update operations.
1178 static ret_code_t write_execute (uint32_t prev_ret, fds_op_t * const p_op)
1179 {
1180  ret_code_t ret;
1181  uint32_t * p_write_addr;
1182  fds_page_t * const p_page = &m_pages[p_op->write.page];
1183  // This must persist across calls.
1184  static fds_record_desc_t desc = {0};
1185  // When a record is updated, this variable will hold the page where the old
1186  // copy was stored. This will be used to set the can_gc flag when the header is
1187  // invalidated (FDS_OP_WRITE_FLAG_DIRTY).
1188  static uint16_t page;
1189 
1190  if (prev_ret != NRF_SUCCESS)
1191  {
1192  // The previous operation has timed out, update offsets.
1193  page_offsets_update (p_page, p_op);
1194  return FDS_ERR_OPERATION_TIMEOUT;
1195  }
1196 
1197  // Compute the address where to write data.
1198  p_write_addr = (uint32_t *) (p_page->p_addr + p_page->write_offset);
1199 
1200  // Execute the current step of the operation, and set one to be executed next.
1201  switch (p_op->write.step)
1202  {
1203  case FDS_OP_WRITE_FIND_RECORD:
1204  {
1205  // The first step of updating a record constists of locating the copy to be deleted.
1206  // If the old copy couldn't be found for any reason then the update should fail.
1207  // This prevents duplicates when queuing multiple updates of the same record.
1208  desc.p_record = NULL;
1209  desc.record_id = p_op->write.record_to_delete;
1210 
1211  if (!record_find_by_desc (&desc, &page))
1212  {
1213  return FDS_ERR_NOT_FOUND;
1214  }
1215 
1216  // Setting the step is redundant since we are falling through.
1217  }
1218 
1219  // Fallthrough to FDS_OP_WRITE_HEADER_BEGIN.
1220 
1221  case FDS_OP_WRITE_HEADER_BEGIN:
1222  ret = record_header_write_begin (p_op, p_write_addr);
1223  break;
1224 
1225  case FDS_OP_WRITE_RECORD_ID:
1226  ret = record_header_write_id (p_op, p_write_addr);
1227  break;
1228 
1229  case FDS_OP_WRITE_DATA:
1230  ret = record_write_data (p_op, p_write_addr);
1231  break;
1232 
1233  case FDS_OP_WRITE_HEADER_FINALIZE:
1234  ret = record_header_write_finalize (p_op, p_write_addr);
1235  break;
1236 
1237  case FDS_OP_WRITE_FLAG_DIRTY:
1238  p_op->write.step = FDS_OP_WRITE_DONE;
1239  ret = record_header_flag_dirty ( (uint32_t *) desc.p_record, page);
1240  break;
1241 
1242  case FDS_OP_WRITE_DONE:
1243  ret = FDS_OP_COMPLETED;
1244 #if (FDS_CRC_CHECK_ON_WRITE)
1245 
1246  if (!crc_verify_success (p_op->write.header.crc16,
1247  p_op->write.header.length_words,
1248  p_write_addr))
1249  {
1250  ret = FDS_ERR_CRC_CHECK_FAILED;
1251  }
1252 
1253 #endif
1254  break;
1255 
1256  default:
1257  ret = FDS_ERR_INTERNAL;
1258  break;
1259  }
1260 
1261  // An operation has either completed or failed. It may have failed because fstorage
1262  // ran out of memory, or because the user tried to delete a record which did not exist.
1263  if (ret != FDS_OP_EXECUTING)
1264  {
1265  // There won't be another callback for this operation, so update the page offset now.
1266  page_offsets_update (p_page, p_op);
1267  }
1268 
1269  return ret;
1270 }

1273 static ret_code_t delete_execute (uint32_t prev_ret, fds_op_t * const p_op)
1274 {
1275  ret_code_t ret;
1276 
1277  if (prev_ret != NRF_SUCCESS)
1278  {
1279  return FDS_ERR_OPERATION_TIMEOUT;
1280  }
1281

1369 static void queue_process (ret_code_t result)
1370 {
1371  static fds_op_t * m_p_cur_op; // Current fds operation.
1372  static nrf_atfifo_item_get_t m_iget_ctx; // Queue context for the current operation.
1373 
1374  while (true)
1375  {
1376  if (m_p_cur_op == NULL)
1377  {
1378  // Load the next from the queue if no operation is being executed.
1379  m_p_cur_op = queue_load (&m_iget_ctx);
1380  }
1381 
1382  /* We can reach here in three ways:
1383  * from queue_start(): something was just queued
1384  * from the fstorage event handler: an operation is being executed
1385  * looping: we only loop if there are operations still in the queue
1386  *
1387  * In all these three cases, m_p_cur_op != NULL.
1388  */
1389  ASSERT (m_p_cur_op != NULL);
1390 
1391  switch (m_p_cur_op->op_code)
1392  {
1393  case FDS_OP_INIT:
1394  result = init_execute (result, m_p_cur_op);
1395  break;
1396 
1397  case FDS_OP_WRITE:
1398  case FDS_OP_UPDATE:
1399  result = write_execute (result, m_p_cur_op);
1400  break;
1401 
1402  case FDS_OP_DEL_RECORD:
1403  case FDS_OP_DEL_FILE:
1404  result = delete_execute (result, m_p_cur_op);
1405  break;
1406 
1407  case FDS_OP_GC:
1408  result = gc_execute (result);
1409  break;
1410 
1411  default:
1412  result = FDS_ERR_INTERNAL;
1413  break;
1414  }
1415 
1416  if (result == FDS_OP_EXECUTING)
1417  {
1418  // The operation has not completed yet. Wait for the next system event.
1419  break;
1420  }
1421 
1422  // The operation has completed (either successfully or with an error).
1423  // - send an event to the user
1424  // - free the operation buffer
1425  // - execute any other queued operations
1426  fds_evt_t evt =
1427  {
1428  // The operation might have failed for one of the following reasons:
1429  // FDS_ERR_BUSY - flash subsystem can't accept the operation
1430  // FDS_ERR_OPERATION_TIMEOUT - flash subsystem timed out
1431  // FDS_ERR_CRC_CHECK_FAILED - a CRC check failed
1432  // FDS_ERR_NOT_FOUND - no record found (delete/update)
1433  .result = (result == FDS_OP_COMPLETED) ? FDS_SUCCESS : result,
1434  };
1435  event_prepare (m_p_cur_op, &evt);
1436  event_send (&evt);
1437  // Zero the pointer to the current operation so that this function
1438  // will fetch a new one from the queue next time it is run.
1439  m_p_cur_op = NULL;
1440  // The result of the operation must be reset upon re-entering the loop to ensure
1441  // the next operation won't be affected by eventual errors in previous operations.
1442  result = NRF_SUCCESS;
1443  // Free the queue element used by the current operation.
1444  queue_free (&m_iget_ctx);
1445 
1446  if (!queue_has_next())
1447  {
1448  // No more elements left. Nothing to do.
1449  break;
1450  }
1451  }
1452 }

1470 // Enqueues write and update operations.
1471 static ret_code_t write_enqueue (fds_record_desc_t * const p_desc,
1472  fds_record_t const * const p_record,
1473  fds_reserve_token_t const * const p_tok,
1474  fds_op_code_t op_code)
1475 {
1476  ret_code_t ret;
1477  uint16_t page;
1478  uint16_t crc = 0;
1479  uint16_t length_words = 0;
1480  fds_op_t * p_op;
1481  nrf_atfifo_item_put_t iput_ctx;
1482 
1483  if (!m_flags.initialized)
1484  {
1485  return FDS_ERR_NOT_INITIALIZED;
1486  }
1487 
1488  if (p_record == NULL)
1489  {
1490  return FDS_ERR_NULL_ARG;
1491  }
1492 
1493  if ( (p_record->file_id == FDS_FILE_ID_INVALID) ||
1494  (p_record->key == FDS_RECORD_KEY_DIRTY))
1495  {
1496  return FDS_ERR_INVALID_ARG;
1497  }
1498 
1499  if (!is_word_aligned (p_record->data.p_data))
1500  {
1501  return FDS_ERR_UNALIGNED_ADDR;
1502  }
1503 
1504  // No space was previously reserved in flash for this operation.
1505  if (p_tok == NULL)
1506  {
1507  // Find a page where to write data.
1508  length_words = p_record->data.length_words;
1509  ret = write_space_reserve (length_words, &page);
1510 
1511  if (ret != FDS_SUCCESS)
1512  {
1513  // There is either not enough space in flash (FDS_ERR_NO_SPACE_IN_FLASH) or
1514  // the record exceeds the size of virtual page (FDS_ERR_RECORD_TOO_LARGE).
1515  return ret;
1516  }
1517  }
1518  else
1519  {
1520  page = p_tok->page;
1521  length_words = p_tok->length_words;
1522  }
1523 
1524  // Get a buffer on the queue of operations.
1525  p_op = queue_buf_get (&iput_ctx);
1526 
1527  if (p_op == NULL)
1528  {
1529  CRITICAL_SECTION_ENTER();
1530  write_space_free (length_words, page);
1531  CRITICAL_SECTION_EXIT();
1532  return FDS_ERR_NO_SPACE_IN_QUEUES;
1533  }
1534 
1535  // Initialize the operation.
1536  p_op->op_code = op_code;
1537  p_op->write.step = FDS_OP_WRITE_HEADER_BEGIN;
1538  p_op->write.page = page;
1539  p_op->write.p_data = p_record->data.p_data;
1540  p_op->write.header.record_id = record_id_new();
1541  p_op->write.header.file_id = p_record->file_id;
1542  p_op->write.header.record_key = p_record->key;
1543  p_op->write.header.length_words = length_words;
1544 
1545  if (op_code == FDS_OP_UPDATE)
1546  {
1547  p_op->write.step = FDS_OP_WRITE_FIND_RECORD;
1548  // Save the record ID of the record to be updated.
1549  p_op->write.record_to_delete = p_desc->record_id;
1550  }
1551 
1552 #if (FDS_CRC_CHECK_ON_READ)
1553  // First, compute the CRC for the first 6 bytes of the header which contain the
1554  // record key, length and file ID, then, compute the CRC of the record ID (4 bytes).
1555  crc = crc16_compute ( (uint8_t *) &p_op->write.header, 6, NULL);
1556  crc = crc16_compute ( (uint8_t *) &p_op->write.header.record_id, 4, &crc);
1557  // Compute the CRC for the record data.
1558  crc = crc16_compute ( (uint8_t *) p_record->data.p_data,
1559  p_record->data.length_words * sizeof (uint32_t), &crc);
1560 #endif
1561  p_op->write.header.crc16 = crc;
1562  queue_buf_store (&iput_ctx);
1563 
1564  // Initialize the record descriptor, if provided.
1565  if (p_desc != NULL)
1566  {
1567  p_desc->p_record = NULL;
1568  // Don't invoke record_id_new() again !
1569  p_desc->record_id = p_op->write.header.record_id;
1570  p_desc->record_is_open = false;
1571  p_desc->gc_run_count = m_gc.run_count;
1572  }
1573 
1574  // Start processing the queue, if necessary.
1575  queue_start();
1576  return FDS_SUCCESS;
1577 }
