530 static rd_status_t ri_bme280_dsp_setup (uint8_t * p_settings_sel, const uint8_t * dsp,
531  uint8_t * parameter)
532 {
533  rd_status_t err_code = RD_SUCCESS;
534 
535  // Error if DSP is not last, and if dsp is something else than IIR or OS
536  if ( (RD_SENSOR_DSP_LAST != *dsp) &&
537  (~ (RD_SENSOR_DSP_LOW_PASS | RD_SENSOR_DSP_OS) & (*dsp)) != 0)
538  {
539  err_code = RD_ERROR_NOT_SUPPORTED;
540  }
541  else
542  {
543  // Always 1x oversampling to keep sensing element enabled
544  dev.settings.osr_h = BME280_OVERSAMPLING_1X;
545  dev.settings.osr_p = BME280_OVERSAMPLING_1X;
546  dev.settings.osr_t = BME280_OVERSAMPLING_1X;
547  dev.settings.filter = BME280_FILTER_COEFF_OFF;
548  *p_settings_sel |= BME280_OSR_PRESS_SEL;
549  *p_settings_sel |= BME280_OSR_TEMP_SEL;
550  *p_settings_sel |= BME280_OSR_HUM_SEL;
551  *p_settings_sel |= BME280_FILTER_SEL;
552 
553  // Setup IIR
554  if (RD_SENSOR_DSP_LOW_PASS & *dsp)
555  {
556  if ( (RD_SENSOR_CFG_DEFAULT == *parameter) || \
557  (RD_SENSOR_CFG_MIN == *parameter) || \
558  (BME280_DSP_MODE_0 == *parameter))
559  {
560  dev.settings.filter = BME280_FILTER_COEFF_OFF;
561  *parameter = BME280_DSP_MODE_0;
562  }
563  else if (BME280_DSP_MODE_1 == *parameter)
564  {
565  dev.settings.filter = BME280_FILTER_COEFF_2;
566  *parameter = BME280_DSP_MODE_1;
567  }
568  else if (BME280_DSP_MODE_2 >= *parameter)
569  {
570  dev.settings.filter = BME280_FILTER_COEFF_4;
571  *parameter = BME280_DSP_MODE_2;
572  }
573  else if (BME280_DSP_MODE_3 >= *parameter)
574  {
575  dev.settings.filter = BME280_FILTER_COEFF_8;
576  *parameter = BME280_DSP_MODE_3;
577  }
578  else if ( (RD_SENSOR_CFG_MAX == *parameter) || \
579  (BME280_DSP_MODE_4 >= *parameter))
580  {
581  dev.settings.filter = BME280_FILTER_COEFF_16;
582  *parameter = BME280_DSP_MODE_4;
583  }
584  else
585  {
586  *parameter = RD_SENSOR_ERR_NOT_SUPPORTED;
587  err_code = RD_ERROR_NOT_SUPPORTED;
588  }
589  }
590 
591  if ( (RD_SUCCESS == err_code) &&
592  ( (RD_SENSOR_DSP_OS & *dsp) != 0))
593  {
594  err_code = ri_bme280_dsp_setup_over (parameter);
595  }
596  }
597 
598  return err_code;

657 {
658  // Assume default / 0,
659  *dsp = RD_SENSOR_CFG_DEFAULT;
660  *parameter = RD_SENSOR_CFG_DEFAULT;
661 
662  // Check if IIR has been set. If yes, read DSP param from there.
663  if (BME280_FILTER_COEFF_OFF != dev.settings.filter)

665  *dsp |= RD_SENSOR_DSP_LOW_PASS;
666 
667  switch (dev.settings.filter)

670  *parameter = BME280_DSP_MODE_1;
671  break;
672

679  break;
680

682  *parameter = BME280_DSP_MODE_4;
683  break;

686  *parameter = BME280_DSP_MODE_1;
687  break;
688  }
689  }
690 
691  // Check if OS has been set. If yes, read DSP param from there.
692  // Param should be same for OS and IIR if it is >1.
693  // OSR is same for every element.
694  if ( (BME280_NO_OVERSAMPLING != dev.settings.osr_h)
695  && (BME280_OVERSAMPLING_1X != dev.settings.osr_h))
696  {
697  *dsp |= RD_SENSOR_DSP_OS;
698 
699  switch (dev.settings.osr_h)
700  {
701  case BME280_OVERSAMPLING_2X:
702  *parameter = BME280_DSP_MODE_1;
703  break;
704 
705  case BME280_OVERSAMPLING_4X:
706  *parameter = BME280_DSP_MODE_2;
707  break;
708 
709  case BME280_OVERSAMPLING_8X:
710  *parameter = BME280_DSP_MODE_3;
711  break;
712 
713  case BME280_OVERSAMPLING_16X:
714  *parameter = BME280_DSP_MODE_4;
715  break;
716 
717  default:
718  *parameter = BME280_DSP_MODE_1;
719  break;
720  }
721  }

746 static rd_status_t ri_bme280_mode_set_single (uint8_t * mode)
747 {
748  rd_status_t err_code = RD_SUCCESS;
749  uint8_t current_mode = RD_SENSOR_CFG_SLEEP;
750  // Do nothing if sensor is in continuous mode
751  ri_bme280_mode_get (&current_mode);
752 
753  if (RD_SENSOR_CFG_CONTINUOUS == current_mode)
754  {
755  *mode = RD_SENSOR_CFG_CONTINUOUS;
756  err_code = RD_ERROR_INVALID_STATE;
757  }
758  else
759  {
760  err_code = BME_TO_RUUVI_ERROR (bme280_set_sensor_mode (BME280_FORCED_MODE, &dev));
761  // We assume that dev struct is in sync with the state of the BME280 and underlying interface
762  // which has the number of settings as 2^OSR is not changed.
763  // We also assume that each element runs same OSR
764  uint8_t samples = (uint8_t) (1U << (dev.settings.osr_h - 1U));
765  ri_delay_ms (bme280_max_meas_time (samples));
766  tsample = rd_sensor_timestamp_get();
767  // BME280 returns to SLEEP after forced sample
768  *mode = RD_SENSOR_CFG_SLEEP;
769  }
770 
771  return err_code;
772 }

854 rd_status_t ri_bme280_data_get (rd_sensor_data_t * const
855  p_data)
856 {
857  rd_status_t err_code = RD_SUCCESS;
858 
859  if (NULL == p_data)
860  {
861  err_code = RD_ERROR_NULL;
862  }
863  else
864  {
865  struct bme280_data comp_data;
866  err_code = BME_TO_RUUVI_ERROR (bme280_get_sensor_data (BME280_ALL, &comp_data, &dev));
867 
868  if (RD_SUCCESS == err_code)
869  {
870  // Write tsample if we're in single mode, current time if we're in continuous mode
871  // Leave sample time as invalid if forced mode is ongoing.
872  uint8_t mode = 0U;
873  err_code |= ri_bme280_mode_get (&mode);
874 
875  if (RD_SENSOR_CFG_SLEEP == mode) { p_data->timestamp_ms = tsample; }
876  else if (RD_SENSOR_CFG_CONTINUOUS == mode) { p_data->timestamp_ms = rd_sensor_timestamp_get(); }
877  else { RD_ERROR_CHECK (RD_ERROR_INTERNAL, ~RD_ERROR_FATAL); }
878 
879  // If we have valid data, return it.
880  if (RD_UINT64_INVALID != p_data->timestamp_ms)
881  {
882  rd_sensor_data_t d_environmental = {0};
883  rd_sensor_data_fields_t env_fields = {.bitfield = 0};
884  float env_values[BME280_SENS_NUM];
885  env_values[BME280_HUMIDITY] = (float) (comp_data.humidity - BME280_HUMIDITY_OFFSET);
886  ri_bme280_check_humiduty (&env_values[BME280_HUMIDITY]);
887  env_values[BME280_PRESSURE] = (float) comp_data.pressure;
888  env_values[BME280_TEMPERATURE] = (float) comp_data.temperature;
889  env_fields.datas.humidity_rh = 1U;
890  env_fields.datas.pressure_pa = 1U;
891  env_fields.datas.temperature_c = 1U;
892  d_environmental.data = env_values;
893  d_environmental.fields = env_fields;
894  d_environmental.valid = env_fields;
895  rd_sensor_data_populate (p_data,
896  &d_environmental,
897  p_data->fields);
898  }
899  }
900  }
901 
902  return err_code;
